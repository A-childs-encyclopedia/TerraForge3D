110void ShadingManager::PrepVertShader()
111{
112 vsh->Clear();
113 vsh->AddTopLine(GLSLLine("layout (location = 0) in vec4 aPos;", "The world position of the vertex"));
114 vsh->AddTopLine(GLSLLine("layout (location = 1) in vec4 aNorm;", "The vertex normals"));
115 vsh->AddTopLine(GLSLLine("layout (location = 2) in vec2 aTexCoord;", "The texture coordinates of the vertex"));
116 vsh->AddTopLine(GLSLLine("layout (location = 3) in vec4 aExtras1;", "Extra data like generated height, erosion delta, etc."));
117 vsh->AddTopLine(GLSLLine("", "The Output passed to the Geometry Shader"));
118 vsh->AddTopLine(GLSLLine(R"(
119out DATA
120{
121 float height;
122 vec3 FragPos;
123 vec3 Normal;
124 vec2 TexCoord;
125} data_out;
126)"));
127
128 vsh->AddUniform(GLSLUniform("_Model", "mat4"));
129 vsh->AddUniform(GLSLUniform("_PV", "mat4"));
130 vsh->AddUniform(GLSLUniform("_TextureBake", "float", "0.0f"));
131 vsh->AddUniform(GLSLUniform("_Scale", "float", "1.0f"));
132
133 GLSLFunction main("main");
134 main.AddLine(GLSLLine("", "TEMP"));
135 main.AddLine(GLSLLine(R"(
136if(_TextureBake == 0.0f)
137 gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0f);
138else
139 gl_Position = vec4(aPos.x/_Scale, aPos.z/_Scale, 0.0f, 1.0f);
140)"));
141 main.AddLine(GLSLLine("data_out.height = aExtras1.x;", "The actual generated noise value"));
142 main.AddLine(GLSLLine("data_out.FragPos = vec3(aPos.x, aPos.y, aPos.z);", "The world position"));
143 main.AddLine(GLSLLine("data_out.Normal = vec3(aNorm.x, aNorm.y, aNorm.z);", "The vertex normals"));
144 main.AddLine(GLSLLine("data_out.TexCoord = aTexCoord;", "The texture coordinates"));
145
146
147 vsh->AddFunction(main);
148}

150void ShadingManager::PrepGeomShader()
151{
152 gsh->Clear();
153
154 gsh->AddTopLine(GLSLLine("layout(triangles) in;", "Take 3 vertices of a triangles as input"));
155 gsh->AddTopLine(GLSLLine("layout(triangle_strip, max_vertices = 3) out;"));
156 gsh->AddTopLine(GLSLLine("", "The data passed in from the vertex shader"));
157 gsh->AddTopLine(GLSLLine(R"(
158in DATA
159{
160 float height;
161 vec3 FragPos;
162 vec3 Normal;
163 vec2 TexCoord;
164} data_in[];
165)"));
166 gsh->AddTopLine(GLSLLine(""));
167 gsh->AddTopLine(GLSLLine("out float height;", "Pass the height to the fragment shader"));
168 gsh->AddTopLine(GLSLLine("out vec4 FragPos;", "Pass the position to the fragment shader"));
169 gsh->AddTopLine(GLSLLine("out vec3 Normal;", "Pass the normal to the fragment shader"));
170 gsh->AddTopLine(GLSLLine("out vec2 TexCoord;", "Pass the texture coordinates to the fragment shader"));
171 gsh->AddTopLine(GLSLLine("out mat3 TBN;", "Pass the Tangent Bitangent Normal matrix to the fragment shader"));
172
173 gsh->AddUniform(GLSLUniform("_PV", "mat4"));
174 gsh->AddUniform(GLSLUniform("_FlatShade", "float"));
175
176 GLSLFunction main("main");
177 main.AddLine(GLSLLine("vec3 a = (_PV * gl_in[0].gl_Position).xyz;", "Vertex a of triangle"));
178 main.AddLine(GLSLLine("vec3 b = (_PV * gl_in[1].gl_Position).xyz;", "Vertex b of triangle"));
179 main.AddLine(GLSLLine("vec3 c = (_PV * gl_in[2].gl_Position).xyz;", "Vertex c of triangle"));
180 main.AddLine(GLSLLine(""));
181 main.AddLine(GLSLLine("vec3 edge0 = b - c;"));
182 main.AddLine(GLSLLine("vec3 edge1 = c - a;"));
183 main.AddLine(GLSLLine(""));
184 main.AddLine(GLSLLine("vec2 deltaUV0 = data_in[1].TexCoord - data_in[0].TexCoord;"));
185 main.AddLine(GLSLLine("vec2 deltaUV1 = data_in[2].TexCoord - data_in[0].TexCoord;"));
186 main.AddLine(GLSLLine(""));
187 main.AddLine(GLSLLine("float invDet = 1.0f / (deltaUV0.x * deltaUV1.y - deltaUV1.x * deltaUV0.y);", "One over the determinant"));
188 main.AddLine(GLSLLine(""));
189 main.AddLine(GLSLLine("vec3 tangent = vec3(invDet * (deltaUV1.y * edge0 - deltaUV0.y * edge1));"));
190 main.AddLine(GLSLLine("vec3 bitangent = vec3(invDet * (-deltaUV1.x * edge0 + deltaUV0.x * edge1));"));
191 main.AddLine(GLSLLine("mat4 model = mat4(1);", "A default model matrix"));
192 main.AddLine(GLSLLine(""));
193 main.AddLine(GLSLLine("vec3 T = normalize(vec3(model * vec4(tangent, 0.0f)));"));
194 main.AddLine(GLSLLine("vec3 B = normalize(vec3(model * vec4(bitangent, 0.0f)));"));
195 main.AddLine(GLSLLine("vec3 N = normalize(vec3(model * vec4(cross(edge1, edge0), 0.0f)));"));
196 main.AddLine(GLSLLine(""));
197 main.AddLine(GLSLLine("TBN = mat3(T, B, N);"));
198 main.AddLine(GLSLLine(""));
199 main.AddLine(GLSLLine("vec3 n = normalize(cross(edge1, edge0));", "Calculate face normal for flat shading"));
200 main.AddLine(GLSLLine(""));
201
202 for(int i=0;i<3;i++)
203 {
204 main.AddLine(GLSLLine("gl_Position = _PV * gl_in[" + std::to_string(i) + "].gl_Position;"));
205 main.AddLine(GLSLLine("height = data_in[" + std::to_string(i) + "].height;"));
206 main.AddLine(GLSLLine("TexCoord = data_in[" + std::to_string(i) + "].TexCoord;"));
207 main.AddLine(GLSLLine("FragPos = vec4(data_in[" + std::to_string(i) + "].FragPos, 1.0f);"));
208 main.AddLine(GLSLLine(R"(
209 if(_FlatShade > 0.5f)
210 Normal = n;
211 else
212 Normal = data_in[)" + std::to_string(i) + R"(].Normal;
213 )"));
214 main.AddLine(GLSLLine("EmitVertex();", "Emit the vertex"));
215}
216
217main.AddLine(GLSLLine("EndPrimitive();", "Emit the triangle"));
218
219
220gsh->AddFunction(main);
221
222}

224void ShadingManager::PrepFragShader()
225{
226 fsh->Clear();
227 fsh->AddTopLine(GLSLLine("out vec4 FragColor;", "The result of the fragment shader"));
228 fsh->AddTopLine(GLSLLine(""));
229 fsh->AddTopLine(GLSLLine("in float height;", "Take the height passed in fromt the Geometry Shader"));
230 fsh->AddTopLine(GLSLLine("in vec4 FragPos;", "Take the world position passed in fromt the Geometry Shader"));
231 fsh->AddTopLine(GLSLLine("in vec3 Normal;", "Take the normal passed in fromt the Geometry Shader"));
232 fsh->AddTopLine(GLSLLine("in vec2 TexCoord;", "Take the texture coordinates passed in fromt the Geometry Shader"));
233 fsh->AddTopLine(GLSLLine(""));
234 fsh->AddTopLine(GLSLLine(R"(
235 struct DataBlob
236 {
237 float d[32];
238 };
239 )"));
240 fsh->AddUniform(GLSLUniform("_LightPosition", "vec3"));
241 fsh->AddUniform(GLSLUniform("_LightColor", "vec3"));
242 fsh->AddUniform(GLSLUniform("_HMapMinMax", "vec3"));
243 fsh->AddUniform(GLSLUniform("_TextureBake", "float", "0.0f"));
244 fsh->AddUniform(GLSLUniform("_Mode", "float", "0.0f"));
245 GLSLSSBO dataBlobs("dataBlobs", std::to_string(sharedMemoryManager->ssboBinding), "These are small data blobs which may be used for nodes, etc.");
246 dataBlobs.AddLine(GLSLLine("DataBlob data[];"));
247 fsh->AddSSBO(dataBlobs);
248 GLSLFunction main("main");
249 main.AddLine(GLSLLine("vec3 objectColor = vec3(1.0f);"));
250 main.AddLine(GLSLLine(""));
251 main.AddLine(GLSLLine(R"(
252 if(_TextureBake == 1.0f)
253 {
254 if(_Mode == 0.0f)
255 objectColor = vec3(1.0);
256 else if(_Mode == 1.0f)
257 objectColor = vec3( (height - _HMapMinMax.x) / (_HMapMinMax.y - _HMapMinMax.x) );
258 else if(_Mode == 2.0f)
259 objectColor = Normal;
260 FragColor = vec4(objectColor, 1.0f);
261 return;
262 }
263 )"));
264 main.AddLine(GLSLLine(""));
265 GLSLLine tmp("", "");
266 NodeInputParam param;
267 param.userData1 = &main;
268 param.userData2 = &tmp;
269 shaderNodeEditor->outputNode->Evaluate(param, nullptr);
270 main.AddLine(GLSLLine("FragColor = vec4(" + tmp.line + ", 1.0f);"));
271 fsh->AddFunction(main);
272}
