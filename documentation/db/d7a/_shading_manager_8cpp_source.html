11static char *stristr4(const char *str, const char *pattern)
12{
13 size_t i;
14
15 if (!*pattern)
16 {
17 return (char *)str;
18 }
19
20 for (; *str; str++)
21 {
22 if (toupper((unsigned char)*str) == toupper((unsigned char)*pattern))
23 {
24 for (i = 1;; i++)
25 {
26 if (!pattern[i])
27 {
28 return (char *)str;
29 }
30
31 if (toupper((unsigned char)str[i]) != toupper((unsigned char)pattern[i]))
32 {
33 break;
34 }
35 }
36 }
37 }
38
39 return NULL;
40}

141void ShadingManager::ReCompileShaders()
142{
143 logs.clear();
144 START_PROFILER();
145
146 sharedMemoryManager->Clear();
147
148 SNENode* outputNode = static_cast<SNENode*>(shaderNodeEditor->outputNode);
149 outputNode->dataBlobOffset = sharedMemoryManager->AddItem();
150 for(auto& it : shaderNodeEditor->nodes)
151 {
152 SNENode* node = static_cast<SNENode*>(it.second);
153 node->dataBlobOffset = sharedMemoryManager->AddItem();
154 }
155
156 outputNode->sharedData = sharedMemoryManager->At(outputNode->dataBlobOffset);
157 outputNode->UpdateShaders();
158 for(auto& it : shaderNodeEditor->nodes)
159 {
160 SNENode* node = static_cast<SNENode*>(it.second);
161 node->sharedData = sharedMemoryManager->At(node->dataBlobOffset);
162 node->UpdateShaders();
163 }
164
165 shaderTextureManager->UpdateShaders();
166
167 PrepVertShader();
168 vertexSource = vsh->GenerateGLSL();
169 PrepGeomShader();
170 geometrySource = gsh->GenerateGLSL();
171 PrepFragShader();
172 fragmentSource = fsh->GenerateGLSL();
173 appState->shaders.terrain = new Shader(vertexSource, fragmentSource, geometrySource);
174 double time = 0.0f;
175 END_PROFILER(time);
176 logs.push_back("Generated & Compiled Shaders in " + std::to_string(time) + "ms");

295void ShadingManager::PrepVertShader()
296{
297 vsh->Clear();
298 vsh->AddTopLine(GLSLLine("layout (location = 0) in vec4 aPos;", "The world position of the vertex"));
299 vsh->AddTopLine(GLSLLine("layout (location = 1) in vec4 aNorm;", "The vertex normals"));
300 vsh->AddTopLine(GLSLLine("layout (location = 2) in vec2 aTexCoord;", "The texture coordinates of the vertex"));
301 vsh->AddTopLine(GLSLLine("layout (location = 3) in vec4 aExtras1;", "Extra data like generated height, erosion delta, etc."));
302 vsh->AddTopLine(GLSLLine("", "The Output passed to the Geometry Shader"));
303 vsh->AddTopLine(GLSLLine(R"(
304out DATA
305{
306 float height;
307 vec3 FragPos;
308 vec3 Normal;
309 vec2 TexCoord;
310} data_out;
311)"));
312
313 vsh->AddUniform(GLSLUniform("_Model", "mat4"));
314 vsh->AddUniform(GLSLUniform("_PV", "mat4"));
315 vsh->AddUniform(GLSLUniform("_TextureBake", "float", "0.0f"));
316 vsh->AddUniform(GLSLUniform("_Scale", "float", "1.0f"));
317
318 GLSLFunction main("main");
319 main.AddLine(GLSLLine("", "TEMP"));
320 main.AddLine(GLSLLine(R"(
321if(_TextureBake == 0.0f)
322 gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0f);
323else
324 gl_Position = vec4(aPos.x/_Scale, aPos.z/_Scale, 0.0f, 1.0f);
325)"));
326 main.AddLine(GLSLLine("data_out.height = aExtras1.x;", "The actual generated noise value"));
327 main.AddLine(GLSLLine("data_out.FragPos = vec3(aPos.x, aPos.y, aPos.z);", "The world position"));
328 main.AddLine(GLSLLine("data_out.Normal = vec3(aNorm.x, aNorm.y, aNorm.z);", "The vertex normals"));
329 main.AddLine(GLSLLine("data_out.TexCoord = aTexCoord;", "The texture coordinates"));
330
331
332 vsh->AddFunction(main);

335void ShadingManager::PrepGeomShader()
336{
337 gsh->Clear();
338
339 gsh->AddTopLine(GLSLLine("layout(triangles) in;", "Take 3 vertices of a triangles as input"));
340 gsh->AddTopLine(GLSLLine("layout(triangle_strip, max_vertices = 3) out;"));
341 gsh->AddTopLine(GLSLLine("", "The data passed in from the vertex shader"));
342 gsh->AddTopLine(GLSLLine(R"(
343in DATA
344{
345 float height;
346 vec3 FragPos;
347 vec3 Normal;
348 vec2 TexCoord;
349} data_in[];
350)"));
351 gsh->AddTopLine(GLSLLine(""));
352 gsh->AddTopLine(GLSLLine("out float height;", "Pass the height to the fragment shader"));
353 gsh->AddTopLine(GLSLLine("out vec4 FragPos;", "Pass the position to the fragment shader"));
354 gsh->AddTopLine(GLSLLine("out vec3 Normal;", "Pass the normal to the fragment shader"));
355 gsh->AddTopLine(GLSLLine("out vec2 TexCoord;", "Pass the texture coordinates to the fragment shader"));
356 gsh->AddTopLine(GLSLLine("out mat3 TBN;", "Pass the Tangent Bitangent Normal matrix to the fragment shader"));
357
358 gsh->AddUniform(GLSLUniform("_PV", "mat4"));
359 gsh->AddUniform(GLSLUniform("_FlatShade", "float"));
360
361 GLSLFunction main("main");
362 main.AddLine(GLSLLine("vec3 a = (_PV * gl_in[0].gl_Position).xyz;", "Vertex a of triangle"));
363 main.AddLine(GLSLLine("vec3 b = (_PV * gl_in[1].gl_Position).xyz;", "Vertex b of triangle"));
364 main.AddLine(GLSLLine("vec3 c = (_PV * gl_in[2].gl_Position).xyz;", "Vertex c of triangle"));
365 main.AddLine(GLSLLine(""));
366 main.AddLine(GLSLLine("vec3 edge0 = b - c;"));
367 main.AddLine(GLSLLine("vec3 edge1 = c - a;"));
368 main.AddLine(GLSLLine(""));
369 main.AddLine(GLSLLine("vec2 deltaUV0 = data_in[1].TexCoord - data_in[0].TexCoord;"));
370 main.AddLine(GLSLLine("vec2 deltaUV1 = data_in[2].TexCoord - data_in[0].TexCoord;"));
371 main.AddLine(GLSLLine(""));
372 main.AddLine(GLSLLine("float invDet = 1.0f / (deltaUV0.x * deltaUV1.y - deltaUV1.x * deltaUV0.y);", "One over the determinant"));
373 main.AddLine(GLSLLine(""));
374 main.AddLine(GLSLLine("vec3 tangent = vec3(invDet * (deltaUV1.y * edge0 - deltaUV0.y * edge1));"));
375 main.AddLine(GLSLLine("vec3 bitangent = vec3(invDet * (-deltaUV1.x * edge0 + deltaUV0.x * edge1));"));
376 main.AddLine(GLSLLine("mat4 model = mat4(1);", "A default model matrix"));
377 main.AddLine(GLSLLine(""));
378 main.AddLine(GLSLLine("vec3 T = normalize(vec3(model * vec4(tangent, 0.0f)));"));
379 main.AddLine(GLSLLine("vec3 B = normalize(vec3(model * vec4(bitangent, 0.0f)));"));
380 main.AddLine(GLSLLine("vec3 N = normalize(vec3(model * vec4(cross(edge1, edge0), 0.0f)));"));
381 main.AddLine(GLSLLine(""));
382 main.AddLine(GLSLLine("TBN = mat3(T, B, N);"));
383 main.AddLine(GLSLLine(""));
384 main.AddLine(GLSLLine("vec3 n = normalize(cross(edge1, edge0));", "Calculate face normal for flat shading"));
385 main.AddLine(GLSLLine(""));
386
387 for(int i=0;i<3;i++)
388 {
389 main.AddLine(GLSLLine("gl_Position = _PV * gl_in[" + std::to_string(i) + "].gl_Position;"));
390 main.AddLine(GLSLLine("height = data_in[" + std::to_string(i) + "].height;"));
391 main.AddLine(GLSLLine("TexCoord = data_in[" + std::to_string(i) + "].TexCoord;"));
392 main.AddLine(GLSLLine("FragPos = vec4(data_in[" + std::to_string(i) + "].FragPos, 1.0f);"));
393 main.AddLine(GLSLLine(R"(
394 if(_FlatShade > 0.5f)
395 Normal = n;
396 else
397 Normal = data_in[)" + std::to_string(i) + R"(].Normal;
398 )"));
399 main.AddLine(GLSLLine("EmitVertex();", "Emit the vertex"));
400}
401
402main.AddLine(GLSLLine("EndPrimitive();", "Emit the triangle"));
403
404
405gsh->AddFunction(main);
406

410{
411 fsh->Clear();

413 fsh->AddTopLine(GLSLLine(""));
414 fsh->AddTopLine(GLSLLine("in float height;", "Take the height passed in fromt the Geometry Shader"));
415 fsh->AddTopLine(GLSLLine("in vec4 FragPos;", "Take the world position passed in fromt the Geometry Shader"));
416 fsh->AddTopLine(GLSLLine("in vec3 Normal;", "Take the normal passed in fromt the Geometry Shader"));
417 fsh->AddTopLine(GLSLLine("in vec2 TexCoord;", "Take the texture coordinates passed in fromt the Geometry Shader"));

420 struct DataBlob

422 float d[32];
423 };
424 )"));
425 fsh->AddUniform(GLSLUniform("_LightPosition", "vec3"));
426 fsh->AddUniform(GLSLUniform("_LightColor", "vec3"));
427 fsh->AddUniform(GLSLUniform("_HMapMinMax", "vec3"));
428 fsh->AddUniform(GLSLUniform("_TextureBake", "float", "0.0f"));
429 fsh->AddUniform(GLSLUniform("_Mode", "float", "0.0f"));

431

433 dataBlobs.AddLine(GLSLLine("DataBlob data[];"));
434 fsh->AddSSBO(dataBlobs);
435 GLSLFunction main("main");
436 main.AddLine(GLSLLine("vec3 objectColor = vec3(1.0f);"));
437 main.AddLine(GLSLLine(""));
438 main.AddLine(GLSLLine(R"(
439 if(_TextureBake == 1.0f)
440 {
441 if(_Mode == 0.0f)
442 objectColor = vec3(1.0);
443 else if(_Mode == 1.0f)
444 objectColor = vec3( (height - _HMapMinMax.x) / (_HMapMinMax.y - _HMapMinMax.x) );
445 else if(_Mode == 2.0f)
446 objectColor = Normal;
447 FragColor = vec4(objectColor, 1.0f);
448 return;
449 }
450 )"));
451 main.AddLine(GLSLLine(""));
452 GLSLLine tmp("", "");
453 NodeInputParam param;
454 param.userData1 = &main;
455 param.userData2 = &tmp;
456 shaderNodeEditor->outputNode->Evaluate(param, nullptr);
457 main.AddLine(GLSLLine("FragColor = vec4(" + tmp.line + ", 1.0f);"));
458 fsh->AddFunction(main);
